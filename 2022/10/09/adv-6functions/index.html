函数改写为前缀形式

x = 10
y = 100
x + y
`+`(x, y)

names(df) <- c("x", "y", "z")
`names<-`(df, c("x", "y", "z"))

for(i in 1:10) print(i)
`for`(i, 1:10, print(i))


lapply(list(1:3, 4:5), `+`, 100)
